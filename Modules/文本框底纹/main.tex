	\def\FrameCommand{%
		\hspace{1pt}%
		{\color{DarkBlue}\vrule width 2pt}%
		{\color{formalshade}\vrule width 4pt}%
		\colorbox{formalshade}%
	}%
	\MakeFramed{\advance\hsize-\width\FrameRestore}%

	}
	{%
		\vspace{2pt}\end{adjustwidth}\endMakeFramed%
}
